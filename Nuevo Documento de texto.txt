letra_inicial="a"
letra_contenida="c"
letra_final="s"

# Construir la expresión regular para buscar palabras
regex="^${letra_inicial}.*${letra_contenida}.*${letra_final}$"

# Ruta al archivo del diccionario (cambia esto según la ubicación de tu diccionario)
diccionario="/ruta/al/diccionario.txt"

# Usar while read para leer el archivo línea por línea
while IFS= read -r palabra; do
    # Verificar si la palabra cumple con la expresión regular
    if [[ "$palabra" =~ $regex ]]; then
        echo "$palabra"
    fi
done < "$diccionario"
